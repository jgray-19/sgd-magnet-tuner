6class LBFGSOptimiser: 

7 """ 

8 Limited-memory BFGS with an adaptive, line-search-free step length. 

9 

10 Adaptive LR: 

11 lr_eff = base_lr * eta_k 

12 eta_k = EMA( clip( (s^T s) / (s^T y), eta_min, eta_max ) ) 

13 

14 Notes 

15 ----- 

16 * Two-loop recursion computes d_k = -H_k g_k; we only scale by lr_eff. 

17 * H0 scaling gamma = (s^T y)/(y^T y) kept (good conditioning). 

18 * BB1 multiplier uses the same (s,y) you already build; no extra evals. 

19 """ 

20 

21 def __init__( 

22 self, 

23 history_size: int = 20, 

24 eps: float = 1e-12, 

25 weight_decay: float = 0.0, 

26 # --- adaptive LR knobs --- 

27 use_adaptive_lr: bool = True, 

28 bb_clip: tuple[float, float] = (1e-2, 1e2), # clip for raw BB1 

29 ema_beta: float = 0.8, # smoothing of BB1 

30 eta_init: float = 1.0, # initial multiplier 

31 ): 

32 self.history_size = int(history_size) 

33 self.eps = float(eps) 

34 self.weight_decay = float(weight_decay) 

35 

36 # State needed to build (s, y) 

37 self.prev_params: np.ndarray | None = None 

38 self.prev_grads: np.ndarray | None = None 

39 

40 # L-BFGS memory 

41 self.S: list[np.ndarray] = [] # s_i = x_{i+1} - x_i 

42 self.Y: list[np.ndarray] = [] # y_i = g_{i+1} - g_i 

43 self.RHO: list[float] = [] # 1 / (y_i^T s_i) 

44 

45 # time step 

46 self.t = 0 

47 

48 # Adaptive LR state 

49 self.use_adaptive_lr = bool(use_adaptive_lr) 

50 self.bb_min, self.bb_max = bb_clip 

51 self.ema_beta = float(ema_beta) 

52 self.eta_ema = float(eta_init)

54 def _push_pair(self, s: np.ndarray, y: np.ndarray) -> None: 

55 ys = float(np.dot(y, s)) 

56 yy = float(np.dot(y, y)) 

57 # Safeguards against near-singular updates 

58 if ys <= 1e-20 or yy <= 1e-20: 

59 return 

60 if len(self.S) == self.history_size: 

61 self.S.pop(0) 

62 self.Y.pop(0) 

63 self.RHO.pop(0) 

64 self.S.append(s) 

65 self.Y.append(y) 

66 self.RHO.append(1.0 / ys)

68 def _two_loop(self, g: np.ndarray) -> np.ndarray: 

69 """Standard two-loop recursion to compute r ≈ H * g.""" 

70 if not self.S: 

71 return g.copy() # H ≈ I at start 

72 

73 q = g.copy() 

74 alpha = [0.0] * len(self.S) 

75 

76 # First loop: descending order 

77 for i in range(len(self.S) - 1, -1, -1): 

78 s_i, y_i, rho_i = self.S[i], self.Y[i], self.RHO[i] 

79 alpha[i] = rho_i * np.dot(s_i, q) 

80 q = q - alpha[i] * y_i 

81 

82 # Initial H0 scaling using the most recent pair (gamma ≈ BB2) 

83 s_last, y_last = self.S[-1], self.Y[-1] 

84 yy = float(np.dot(y_last, y_last)) 

85 ys = float(np.dot(y_last, s_last)) 

86 gamma = ys / yy if yy > 0 else 1.0 

87 r = gamma * q 

88 

89 # Second loop: ascending order 

90 for i in range(len(self.S)): 

91 s_i, y_i, rho_i = self.S[i], self.Y[i], self.RHO[i] 

92 beta = rho_i * np.dot(y_i, r) 

93 r = r + s_i * (alpha[i] - beta) 

94 return r

96 def _bb1_multiplier(self, s: np.ndarray, y: np.ndarray) -> float: 

97 """ 

98 BB1 spectral step (s^T s)/(s^T y); clipped + EMA-smoothed. 

99 Use only when we have a fresh pair. 

100 """ 

101 denom = float(np.dot(s, y)) 

102 if denom <= self.eps: 

103 return self.eta_ema # keep previous multiplier if curvature poor 

104 raw = float(np.dot(s, s)) / denom 

105 # clip to avoid outliers, then EMA 

106 raw = min(max(raw, self.bb_min), self.bb_max) 

107 self.eta_ema = self.ema_beta * self.eta_ema + (1.0 - self.ema_beta) * raw 

108 return self.eta_ema

110 def step(self, params: np.ndarray, grads: np.ndarray, lr: float) -> np.ndarray: 

111 """ 

112 Perform one L-BFGS step with adaptive LR multiplier. 

113 

114 Args 

115 ---- 

116 params : np.ndarray 

117 Current parameter vector x_k. 

118 grads : np.ndarray 

119 Current gradient g_k = ∇f(x_k). 

120 lr : float 

121 Base learning rate (e.g. from your cosine scheduler). 

122 """ 

123 self.t += 1 

124 

125 # Weight decay (L2 regularisation) 

126 g = grads + (self.weight_decay * params if self.weight_decay != 0 else 0) 

127 

128 # On k ≥ 1, build (s_{k-1}, y_{k-1}) from the PREVIOUS state 

129 eta_mult = self.eta_ema 

130 if self.prev_params is not None and self.prev_grads is not None: 

131 s = params - self.prev_params 

132 y = g - self.prev_grads 

133 self._push_pair(s, y) 

134 if self.use_adaptive_lr: 

135 eta_mult = self._bb1_multiplier(s, y) 

136 

137 # Compute quasi-Newton direction: d = - H_k * g_k 

138 Hg = self._two_loop(g) # noqa: N806 

139 d = -Hg 

140 

141 # Take step with adaptive multiplier (no line search) 

142 lr_eff = lr * eta_mult 

143 new_params = params + lr_eff * d 

144 

145 # Save state for the next call 

146 self.prev_params = params.copy() 

147 self.prev_grads = g.copy() 

148 

149 return new_params
